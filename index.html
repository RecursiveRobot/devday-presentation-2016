Function stored in Variable


var double = function(x) {
  x * 2;
};

double(5);
// 10

Function as Parameter


var double = x => x * 2;

function apply(func, value) {
  return func(value);
};

apply(double, 5);
// 10

// sumList :: [int] -> int
let rec sumList xs =
    match xs with
    | []    -> 0
    | y::ys -> y + sumList ys

Addition


// Uncurried..
function add(a, b) {
    return a + b;
}

var c = add(1, 2);
// 3

Addition


// Curried...
function addCurried(a) {
    return function(b) {
        return a + b;
    }
}

var d = addCurried(1)(2);
// 3

var add5 = addCurried(5);
// add5 :: int -> int
var e = add5(2);
// 7
